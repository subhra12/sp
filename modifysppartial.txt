ALTER PROCEDURE [dbo].[sp_UnApprovedHours_PM] 
@Emp_ID INT=null, 
@sDate Datetime,                          
@eDate Datetime
AS                              
BEGIN						          

IF OBJECT_ID(N'tempdb..#TempDates') IS NOT NULL
BEGIN
DROP TABLE #TempDates
END

IF OBJECT_ID(N'tempdb..#TempDetails') IS NOT NULL
BEGIN
DROP TABLE #TempDetails
END


select  @eDate = case when @eDate is null then GETDATE() else @eDate end

;with cte as (
  select convert(date,@sDate) startDate,
         month(@sdate) n
  union all
  select dateadd(month,n,convert(date,convert(varchar,year(@sdate)) + '0101')) startDate,
        (n+1) n
  from cte
  where n < month(@sdate) + datediff(month,@sdate,@edate)
)
SELECT * into #TempDates from
(select startdate,cast (case when Month(startDate)=Month(@eDate)  and year(startDate)=year(@eDate) then @eDate else 
DATEADD(d, -1, DATEADD(m, DATEDIFF(m, 0, startDate) + 1, 0))  end as date) enddate
from cte ) as x

--select * from #TempDates

CREATE TABLE #TempDetails (
	ProjectNo varchar(50),
	ProjectName varchar(50),
	PM varchar(50),
	PMEmployeecode varchar(50),
	Month varchar(10),
	EmployeeName varchar(50),
	EmployeeCode varchar(50),
	UnapprovedHours decimal(18,2),
)

DECLARE Cur_1 CURSOR
    FOR SELECT * FROM #TempDates


OPEN Cur_1
DECLARE @StartDATE date
DECLARE @EndDATE date
FETCH NEXT FROM Cur_1 
INTO @StartDATE, @EndDATE

    WHILE (@@FETCH_STATUS = 0)
        BEGIN
		insert into #tempdetails  
		SElECT  tb.ProjectNo,BD.ProjectDescription,

		case when BD.WBSPersonResponsible is null or BD.WBSPersonResponsible=0 then BD.ProjectPersonResponsibleName
		else BD.WBSPersonResponsibleName end as WBSPersonResponsibleName,
		
		case when BD.WBSPersonResponsible is null or BD.WBSPersonResponsible=0 then BD.ProjectPersonResponsible
		else BD.WBSPersonResponsible end as WBSPersonResponsiblecode,

		LEFT(DATENAME(MONTH, @StartDATE), 3)+'-'+Convert(varchar,Year(@StartDATE)) as 'Month',tb.EmployeeName,tb.Emp_Id,
		sum(tbd.BookedHours) as DHUnapproved	
		from TRN_TimeBooking tb  
		inner join TRN_BookedDate tbd on tb.TimeBooking_Id= tbd.TimeBooking_Id
		inner join Budget BD on BD.Projectcode=tb.ProjectNo and bd.WBSnumber=tb.WBSNo
		where  
		cast(tbd.Date as date) BETWEEN  @StartDATE  AND  @EndDATE
		and tb.issubmitted=1 and tb.PMApproval=0 and tb.dhapproval=1 and isrejected=0
		and (@Emp_ID is null or  (((BD.WBSPersonResponsible IS NULL OR BD.WBSPersonResponsible = 0) AND BD.ProjectPersonResponsible = @Emp_ID )
         OR (BD.WBSPersonResponsible = @Emp_ID )))
		--BD.WBSPersonResponsible= CASE WHEN @Emp_ID IS NULL THEN BD.WBSPersonResponsible ELSE @Emp_ID END
		group by tb.ProjectNo,BD.ProjectDescription,
		case when BD.WBSPersonResponsible is null or BD.WBSPersonResponsible=0 then BD.ProjectPersonResponsibleName
		else BD.WBSPersonResponsibleName end ,
		case when BD.WBSPersonResponsible is null or BD.WBSPersonResponsible=0 then BD.ProjectPersonResponsible
		else BD.WBSPersonResponsible end ,
		tb.EmployeeName,tb.Emp_Id

	FETCH NEXT FROM Cur_1 INTO @StartDATE, @EndDATE
        END
CLOSE Cur_1

select * from #TempDetails order by EmployeeName,Month
DEALLOCATE Cur_1


END

CREATE procedure [dbo].[EMPLOYEESTATUSDRAFT24092025]

@FromDate Datetime,                          
@ToDate Datetime

As
begin
SET NOCOUNT ON;

if (@todate>GETDATE())
	begin
		set @todate=GETDATE()
	end

IF OBJECT_ID('tempdb..#Stage') IS NOT NULL DROP TABLE #Stage;
IF OBJECT_ID('tempdb..#TimeData') IS NOT NULL DROP TABLE #TimeData;
;WITH WeekPeriods AS (
    SELECT DISTINCT 
        FromDate, 
        ToDate, 
        FORMAT(FromDate, 'MMMM yyyy') AS [Month]
    FROM dbo.TRN_TimeBooking
),
WeekDays AS (
    -- Generate all weekdays (Mon–Fri) between FromDate and ToDate
    SELECT 
        wp.FromDate,
        wp.ToDate,
        d.Date
    FROM WeekPeriods wp
    CROSS APPLY (
        SELECT DATEADD(DAY, v.number, wp.FromDate) AS Date
        FROM master.dbo.spt_values v
        WHERE v.type = 'P'
          AND DATEADD(DAY, v.number, wp.FromDate) <= wp.ToDate
          AND DATENAME(WEEKDAY, DATEADD(DAY, v.number, wp.FromDate)) NOT IN ('Saturday','Sunday')
    ) d
),
WeeklyStatus AS (
    SELECT
        tb.Emp_ID,
        tb.FromDate,
        tb.ToDate,
        SUM(ISNULL(bd.BookedHours, 0)) AS TotalHours,
        SUM(CASE 
                WHEN ISNULL(tb.PMApproval, 0) = 1 
                     AND ISNULL(tb.DHApproval, 0) = 1 
                THEN ISNULL(bd.BookedHours, 0) 
                ELSE 0 
            END) AS ApprovedHours
    FROM dbo.TRN_TimeBooking tb
    JOIN dbo.TRN_BookedDate bd 
        ON tb.TimeBooking_Id = bd.TimeBooking_Id
    GROUP BY tb.Emp_ID, tb.FromDate, tb.ToDate
)


SELECT
    mu.EmpNo AS Emp_ID,
    tb.TimeBooking_Id AS TimesheetNo,
    dhUser.EmployeeName AS ApproverHHName,
    pmUser.EmployeeName AS ApproverPMName,
    mu.EmployeeName AS EmployeeName,
    mu.UserType AS EmployeeType,
    mu.EmpDept AS Department,
    CAST(wp.FromDate AS DATE) as FromDate,
    CAST(wp.ToDate AS DATE) AS ToDate,
    CAST(wd.Date AS DATE) AS TransactionDate,
    bd.BookedHours,
    tb.ProjectNo,
    pb.ProjectDescription AS ProjectDescription,
    tb.WBSNo,
    pb.WBSDescription,
    tb.PO,
    CAST(tb.BookedOn AS DATE) AS BookedOn,
    lr.Rate AS [LabourRate],
    CASE 
        WHEN tb.TimeBooking_Id IS NULL THEN 'Not Booked'
        WHEN ws.TotalHours > 0 AND ws.ApprovedHours >= ws.TotalHours AND bd.BookedHours > 0
             THEN 'Approved'
        WHEN ws.TotalHours > 0 AND ws.ApprovedHours < ws.TotalHours 
             AND tb.DHApproval = 1 AND tb.PMApproval = 1 AND bd.BookedHours > 0
             THEN 'Partially Approved'
        WHEN ws.TotalHours > 0 AND ws.ApprovedHours < ws.TotalHours 
             AND tb.DHApproval = 0 AND tb.IsSubmitted = 1 AND bd.BookedHours > 0
             THEN 'Pending with HH'
        WHEN ws.TotalHours > 0 AND ws.ApprovedHours < ws.TotalHours 
             AND tb.DHApproval = 1 AND (tb.PMApproval IS NULL OR tb.PMApproval = 0) AND bd.BookedHours > 0
             THEN 'Pending with PM'
        WHEN tb.IsRejected = 1 AND ISNULL(tb.PMRemark,'') <> '' THEN 'PM rejected'
        WHEN tb.IsRejected = 1 AND ISNULL(tb.DHRemark,'') <> '' THEN 'HH rejected'
        WHEN tb.IsSubmitted = 0 THEN 'Draft'
        ELSE 'Unknown'
    END AS Status,
    CAST(GETDATE() AS DATE) AS ReportDate,
    FORMAT(GETDATE(), 'hh:mm tt') AS ReportTime,
    wp.Month AS Month,
	--ADD BILLABLE TYPE
	CASE WHEN ISNULL(lr.Rate,0)>0 THEN 'Billable' ELSE 'Non-Billable' END AS BillableType
INTO #Stage
FROM dbo.MST_User mu
CROSS JOIN WeekPeriods wp
JOIN WeekDays wd ON wd.FromDate = wp.FromDate AND wd.ToDate = wp.ToDate
LEFT JOIN dbo.TRN_TimeBooking tb 
    ON mu.EmpNo = tb.Emp_ID AND tb.FromDate = wp.FromDate AND tb.ToDate = wp.ToDate
LEFT JOIN dbo.TRN_BookedDate bd 
    ON tb.TimeBooking_Id = bd.TimeBooking_Id AND bd.Date = wd.Date
	--And tb.FromDate BETWEEN  @FromDate  AND  @ToDate
LEFT JOIN dbo.Budget pb ON tb.ProjectNo = pb.Projectcode AND tb.WBSNo = pb.WBSNumber
LEFT JOIN dbo.MST_User dhUser ON tb.ApprovedByDH = dhUser.EmpNo
LEFT JOIN dbo.MST_User pmUser ON tb.ApprovedByPM = pmUser.EmpNo
LEFT JOIN dbo.MST_LabourRate lr 
    ON lr.Department = mu.EmpDept AND lr.EmployeeType = mu.UserType AND lr.Grade = mu.EmpGrade
   AND tb.BookedOn >= lr.FromDate AND (lr.ToDate IS NULL OR tb.BookedOn <= lr.ToDate)
LEFT JOIN WeeklyStatus ws ON ws.Emp_ID = tb.Emp_ID AND ws.FromDate = tb.FromDate AND ws.ToDate = tb.ToDate
WHERE CAST(wd.Date AS DATE)>=CAST(mu.CreatedOn AS DATE) and (mu.InactiveDate IS NULL or mu.InactiveDate >=CAST(wd.Date AS DATE)) --And tb.FromDate BETWEEN  @FromDate  AND  @ToDate--AND mu.EmpNo='950228';

truncate table timedata

INSERT INTO TimeData
SELECT *
FROM #Stage
WHERE BookedHours > 0 AND Status <> 'Unknown';

INSERT INTO TimeData
SELECT 
    s.Emp_ID,
    NULL AS TimesheetNo,
    NULL AS ApproverHHName,
    NULL AS ApproverPMName,
    s.EmployeeName,
    s.EmployeeType,
    s.Department,
    s.FromDate,
    s.ToDate,
    s.TransactionDate,
    8 AS BookedHours,
    NULL AS ProjectNo,
    NULL AS ProjectDescription,
    NULL AS WBSNo,
    NULL AS WBSDescription,
    NULL AS PO,
    NULL AS BookedOn,
    (select top 1 lr.Rate from dbo.MST_LabourRate lr left join MST_User mu
    ON  lr.Department = mu.EmpDept AND lr.EmployeeType = mu.UserType AND lr.Grade = mu.EmpGrade 
	AND(s.TransactionDate >= lr.FromDate AND (lr.ToDate IS NULL OR s.TransactionDate <= lr.ToDate))
	where mu.EmpNo=s.Emp_ID
   ) AS LabourRate,
    'Not Booked' AS Status,
    CAST(GETDATE() AS DATE) AS ReportDate,
    FORMAT(GETDATE(), 'hh:mm tt') AS Reportime,
    s.Month,
	CASE WHEN (select top 1 lr.Rate from dbo.MST_LabourRate lr left join MST_User mu
    ON  lr.Department = mu.EmpDept AND lr.EmployeeType = mu.UserType AND lr.Grade = mu.EmpGrade 
	AND(s.TransactionDate >= lr.FromDate AND (lr.ToDate IS NULL OR s.TransactionDate <= lr.ToDate))
	where mu.EmpNo=s.Emp_ID
   )>0 THEN 'Billable' ELSE 'Non-Billable' END AS BillableType
FROM (
    SELECT DISTINCT Emp_ID, EmployeeName, EmployeeType, Department,
           FromDate, ToDate, TransactionDate, Month,BillableType
    FROM #Stage
) s
WHERE NOT EXISTS (
    SELECT 1
    FROM TimeData td
    WHERE td.Emp_ID = s.Emp_ID
      AND CAST(td.TransactionDate AS DATE) = CAST(s.TransactionDate AS DATE)
);

SELECT * FROM TimeData where TransactionDate BETWEEN @FromDate AND @ToDate and EmployeeType in('Onroll','Offroll') and LabourRate>=0 
AND NOT(EmployeeType='Offroll' and BillableType='Non-Billable')
--@FromDate  AND  @ToDate

ORDER BY EmployeeName, FromDate, TransactionDate;
END
