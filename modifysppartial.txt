WHEN tb.TimeBooking_Id IS NULL 
         AND NOT EXISTS (
              SELECT 1
              FROM dbo.TRN_TimeBooking tb2
              JOIN dbo.TRN_BookedDate bd2 
                   ON tb2.TimeBooking_Id = bd2.TimeBooking_Id
              WHERE tb2.Emp_ID = mu.EmpNo
                AND bd2.[Date] = wd.[Date]
                AND bd2.BookedHours > 0
          )
    THEN 'Not Booked'
