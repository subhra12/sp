
ALTER PROCEDURE [dbo].[EMPLOYEESTATUSDRAFT24092025]
(
    @FromDate DATETIME,
    @ToDate   DATETIME
)
AS
BEGIN
    SET NOCOUNT ON;

    IF (@ToDate > GETDATE())
        SET @ToDate = GETDATE();

    IF OBJECT_ID('tempdb..#Stage') IS NOT NULL DROP TABLE #Stage;
    IF OBJECT_ID('tempdb..#TimeData') IS NOT NULL DROP TABLE #TimeData;

    /* =========================================================
       WEEK RANGE GENERATION
    ========================================================= */

    ;WITH WeekPeriods AS (
        SELECT DISTINCT 
            FromDate,
            ToDate,
            FORMAT(FromDate, 'MMMM yyyy') AS [Month]
        FROM dbo.TRN_TimeBooking
        WHERE FromDate BETWEEN @FromDate AND @ToDate
    ),
    WeekDays AS (
        SELECT 
            wp.FromDate,
            wp.ToDate,
            d.Date
        FROM WeekPeriods wp
        CROSS APPLY (
            SELECT DATEADD(DAY, v.number, wp.FromDate) AS Date
            FROM master.dbo.spt_values v
            WHERE v.type = 'P'
              AND DATEADD(DAY, v.number, wp.FromDate) <= wp.ToDate
              AND DATENAME(WEEKDAY, DATEADD(DAY, v.number, wp.FromDate)) 
                    NOT IN ('Saturday','Sunday')
        ) d
    ),
    WeeklyStatus AS (
        SELECT
            tb.Emp_ID,
            tb.FromDate,
            tb.ToDate,
            SUM(ISNULL(bd.BookedHours,0)) AS TotalHours,
            SUM(CASE 
                    WHEN ISNULL(tb.PMApproval,0)=1 
                     AND ISNULL(tb.DHApproval,0)=1 
                    THEN ISNULL(bd.BookedHours,0)
                    ELSE 0
                END) AS ApprovedHours
        FROM dbo.TRN_TimeBooking tb
        JOIN dbo.TRN_BookedDate bd
            ON tb.TimeBooking_Id = bd.TimeBooking_Id
        GROUP BY tb.Emp_ID, tb.FromDate, tb.ToDate
    )

    /* =========================================================
       MAIN DATA SET
    ========================================================= */

    SELECT
        mu.EmpNo AS Emp_ID,
        tb.TimeBooking_Id AS TimesheetNo,
        dhUser.EmployeeName AS ApproverHHName,
        pmr.EffectivePMName AS ApproverPMName,  -- ✅ Correct PM name
        mu.EmployeeName,
        mu.UserType AS EmployeeType,
        mu.EmpDept AS Department,
        CAST(wp.FromDate AS DATE) AS FromDate,
        CAST(wp.ToDate AS DATE) AS ToDate,
        CAST(wd.Date AS DATE) AS TransactionDate,
        bd.BookedHours,
        tb.ProjectNo,
        pb.ProjectDescription,
        tb.WBSNo,
        pb.WBSDescription,
        tb.PO,
        CAST(tb.BookedOn AS DATE) AS BookedOn,
        lr.Rate AS LabourRate,

        /* ================= STATUS LOGIC ================= */

        CASE 
            WHEN tb.TimeBooking_Id IS NULL THEN 'Not Booked'

            WHEN ws.TotalHours > 0 
             AND ws.ApprovedHours >= ws.TotalHours 
             AND bd.BookedHours > 0
                THEN 'Approved'

            WHEN ws.TotalHours > 0 
             AND ws.ApprovedHours < ws.TotalHours
             AND tb.DHApproval = 1 
             AND tb.PMApproval = 1
             AND bd.BookedHours > 0
                THEN 'Partially Approved'

            WHEN ws.TotalHours > 0 
             AND ws.ApprovedHours < ws.TotalHours
             AND tb.DHApproval = 0
             AND tb.IsSubmitted = 1
             AND bd.BookedHours > 0
                THEN 'Pending with HH'

            /* ✅ PM PENDING LOGIC FROM sp_UnApprovedHours_PM */
            WHEN ws.TotalHours > 0 
             AND ws.ApprovedHours < ws.TotalHours
             AND tb.DHApproval = 1
             AND (tb.PMApproval IS NULL OR tb.PMApproval = 0)
             AND pmr.EffectivePMEmpID IS NOT NULL
             AND bd.BookedHours > 0
                THEN 'Pending with PM'

            WHEN tb.IsRejected = 1 
             AND ISNULL(tb.PMRemark,'') <> ''
                THEN 'PM rejected'

            WHEN tb.IsRejected = 1 
             AND ISNULL(tb.DHRemark,'') <> ''
                THEN 'HH rejected'

            WHEN tb.IsSubmitted = 0
                THEN 'Draft'

            ELSE 'Unknown'
        END AS Status,

        CAST(GETDATE() AS DATE) AS ReportDate,
        FORMAT(GETDATE(), 'hh:mm tt') AS ReportTime,
        wp.Month,

        CASE 
            WHEN ISNULL(lr.Rate,0) > 0 
                THEN 'Billable' 
            ELSE 'Non-Billable' 
        END AS BillableType
