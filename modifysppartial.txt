JOIN WeekPeriods wp
   ON EXISTS (
        SELECT 1 
        FROM dbo.TRN_TimeBooking t
        WHERE t.Emp_ID = mu.EmpNo
          AND t.FromDate = wp.FromDate
          AND t.ToDate   = wp.ToDate
    )
   OR NOT EXISTS (
        SELECT 1 
        FROM dbo.TRN_TimeBooking t
        WHERE t.Emp_ID = mu.EmpNo
          AND t.FromDate = wp.FromDate
          AND t.ToDate   = wp.ToDate
    )
